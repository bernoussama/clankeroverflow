Resolution Strategy

By default, Cloudflare Pages deployments cache static assets and fetch responses aggressively. When using Next.js on Pages, the standard revalidateTag() needs explicit execution context propagation...

// apps/server/src/index.ts or Next.js route
export async function POST(req: Request) {
  const { tag } = await req.json();

  // Under Cloudflare, waitUntil keeps the worker active
  ctx.waitUntil(
    fetch(`https://api.cloudflare.com/client/v4/zones/${ZONE_ID}/purge_cache`, {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${CF_API_TOKEN}`,
        'Content-Type': 'application/json'
      },
      body: JSON.stringify({ tags: [tag] })
    })
  );

  return Response.json({ purged: true });
}
